019package org.javamoney.cdi.internal;
020
021import javax.money.spi.ServiceProvider;
022
023import java.util.ArrayList;
024import java.util.Collections;
025import java.util.List;
026import java.util.ServiceLoader;
027import java.util.concurrent.ConcurrentHashMap;
028import java.util.logging.Level;
029import java.util.logging.Logger;
030
031/**
032 * Default SE ServiceLoader based ServiceProvider implementation.
033 */
034public class DefaultServiceProvider implements ServiceProvider {
035    /**
036     * List of services loaded, per class.
037     */
038    private final ConcurrentHashMap<Class, List<Object>> servicesLoaded = new ConcurrentHashMap<>();
039
040    /**
041     * Loads and registers services.
042     *
043     * @param serviceType The service type.
044     * @param <T>         the concrete type.
045     * @param defaultList the list of items returned, if no services were found.
046     * @return the items found, never {@code null}.
047     */
048    public <T> List<T> getServices(final Class<T> serviceType, final List<T> defaultList) {
049        @SuppressWarnings("unchecked")
050        List<T> found = (List<T>) servicesLoaded.get(serviceType);
051        if (found != null) {
052            return found;
053        }
054
055        return loadServices(serviceType, defaultList);
056    }
057
058    /**
059     * Loads and registers services.
060     *
061     * @param serviceType The service type.
062     * @param <T>         the concrete type.
063     * @param defaultList the list of items returned, if no services were found.
064     * @return the items found, never {@code null}.
065     */
066    private <T> List<T> loadServices(final Class<T> serviceType, final List<T> defaultList) {
067        try {
068            List<T> services = new ArrayList<>();
069            for (T t : ServiceLoader.load(serviceType)) {
070                services.add(t);
071            }
072            if (services.isEmpty()) {
073                services.addAll(defaultList);
074            }
075            @SuppressWarnings("unchecked")
076            final List<T> previousServices = (List<T>) servicesLoaded.putIfAbsent(serviceType, (List<Object>) services);
077            return Collections.unmodifiableList(previousServices != null ? previousServices : services);
078        } catch (Exception e) {
079            Logger.getLogger(DefaultServiceProvider.class.getName()).log(Level.WARNING,
080                    "Error loading services of type " + serviceType, e);
081            return defaultList;
082        }
083    }
084
085        @Override
086        public int getPriority() {
087                return 0;
088        }
089
090        @Override
091        public <T> List<T> getServices(Class<T> serviceType) {
092                // TODO Auto-generated method stub
093                return null;
094        }
095
096}
